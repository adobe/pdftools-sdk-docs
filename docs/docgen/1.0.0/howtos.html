    //Get the samples from https://www.adobe.com/go/pdftoolsapi_java_samples
  package com.adobe.platform.operation.samples.documentmerge;

  public class MergeDocumentToDOCX {
     public static void main(String[] args) {

         try {
           // Create a context
           ExecutionContext executionContext = SampleTestUtils.createExecutionContext();

           JSONObject jsonDataForMerge=new JSONObject("{\"customerName\": \"James\",\"customerVisits\": 100}");

           // Create a new DocumentMergeOptions instance with required parameters
           DocumentMergeOptions documentMergeOptions = new DocumentMergeOptions(jsonDataForMerge, OutputFormat.DOCX);

           // Create a new DocumentMergeOperation instance with the DocumentMergeOptions instance
           DocumentMergeOperation documentMergeOperation = DocumentMergeOperation.createNew(documentMergeOptions);

           // Set the operation input i.e. the document template
           FileRef documentTemplate = FileRef.createFromLocalFile("src/test/resources/documentTemplate.docx");
           documentMergeOperation.setInput(documentTemplate);

           // Execute the operation
           FileRef result = documentMergeOperation.execute(executionContext);

           // Save the result at the specified location
           result.saveAs("output/documentMergeOutput.docx");
         } catch (ServiceApiException | IOException | ServiceUsageException e) {
           System.out.println(e);
         }
     }
  }

    // Get the samples from http://www.adobe.com/go/pdftoolsapi_node_sample
 const PDFToolsSdk = require('@adobe/documentservices-pdftools-node-sdk');

 try {
     // Initial setup, create credentials instance.
     const credentials =  PDFToolsSdk.Credentials
       .serviceAccountCredentialsBuilder()
       .fromFile("pdftools-api-credentials.json")
       .build();

     //Create an ExecutionContext using credentials and create a new operation instance.
     const clientConfig = PDFToolsSdk.ClientConfig
                   .clientConfigBuilder()
                   .fromFile("pdftools-api-client-config.json")
                   .build();

     const clientContext = PDFToolsSdk.ExecutionContext
                   .create(credentials,clientConfig);

     const jsonDataForMerge = JSON.parse("{\"customerName\": \"James\", \"customerVisits\": 100}");

     let documentMergeOptions = new PDFToolsSdk.DocumentMerge.options.DocumentMergeOptions(jsonDataForMerge,PDFToolsSdk.DocumentMerge.options.OutputFormat.DOCX);

     const documentMergeOperation = PDFToolsSdk.DocumentMerge.Operation.createNew(documentMergeOptions),
                   input = PDFToolsSdk.FileRef.createFromLocalFile('test/resources/documentTemplate.docx',PDFToolsSdk.DocumentMerge.SupportedSourceFormat.docx);
     documentMergeOperation.setInput(input);

     // Execute the operation and Save the result to the specified location.
     documentMergeOperation.execute(clientContext)
                   .then(result => result.saveAsFile('output/documentMergeOutput.docx'))
                   .catch(err => console.log(err));
   } catch (err) {
     console.log('Exception encountered while executing operation', err);
   }
